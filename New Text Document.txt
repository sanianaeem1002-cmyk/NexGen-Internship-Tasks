#include <iostream>
#include <iomanip> // For controlling decimal precision
#include <cmath>   // For advanced math if needed later

using namespace std;

void showMenu();
void runCalculator();
void checkNumber();
void calculateGrade();

int main() {
    int choice;
    // Set output to show 2 decimal places for a "clean" look
    cout << fixed << setprecision(2);

    while (true) {
        showMenu();
        if (!(cin >> choice)) { // Error handling for non-numeric input
            cout << "Invalid input. Please enter a number." << endl;
            cin.clear();
            while (cin.get() != '\n');
            continue;
        }

        switch (choice) {
            case 1: runCalculator(); break;
            case 2: checkNumber(); break;
            case 3: calculateGrade(); break;
            case 4: 
                cout << "\nExiting... Good luck with your internship!" << endl;
                return 0;
            default: cout << "Invalid selection." << endl;
        }
    }
    return 0;
}

void showMenu() {
    cout << "\n================================" << endl;
    cout << "   NEXGEN UTILITY TOOL (V2.0)" << endl;
    cout << "================================" << endl;
    cout << "1. Advanced Calculator (+, -, *, /)" << endl;
    cout << "2. Number Property Checker" << endl;
    cout << "3. Grading System" << endl;
    cout << "4. Exit" << endl;
    cout << "Select option: ";
}

void runCalculator() {
    double num1, num2, result;
    char op;

    cout << "\n[ Calculator Mode ]" << endl;
    cout << "Supports negative values (e.g., -5) and decimals (e.g., 2.5)" << endl;
    cout << "Enter expression (e.g., 5 - 10 or -2.5 * 4): ";
    cin >> num1 >> op >> num2;

    switch (op) {
        case '+': result = num1 + num2; break;
        case '-': result = num1 - num2; break;
        case '*': result = num1 * num2; break;
        case '/': 
            if (num2 != 0) {
                result = num1 / num2;
            } else {
                cout << "Error: Cannot divide by zero!" << endl;
                return;
            }
            break;
        default:
            cout << "Error: Invalid operator!" << endl;
            return;
    }

    // This handles negative results and decimals automatically
    cout << "RESULT: " << result << endl;
}

void checkNumber() {
    double input;
    cout << "\nEnter a number: ";
    cin >> input;

    // Convert to int for parity/prime checks
    int num = static_cast<int>(input);

    if (num % 2 == 0) cout << num << " is Even." << endl;
    else cout << num << " is Odd." << endl;

    bool isPrime = (num > 1);
    for (int i = 2; i <= sqrt(num); i++) {
        if (num % i == 0) {
            isPrime = false;
            break;
        }
    }
    if (isPrime) cout << num << " is also a Prime number." << endl;
}

void calculateGrade() {
    double score;
    cout << "\nEnter score (0-100): ";
    cin >> score;

    if (score > 100 || score < 0) cout << "Invalid score!" << endl;
    else if (score >= 90) cout << "Grade: A" << endl;
    else if (score >= 80) cout << "Grade: B" << endl;
    else if (score >= 70) cout << "Grade: C" << endl;
    else if (score >= 60) cout << "Grade: D" << endl;
    else cout << "Grade: F" << endl;
}
